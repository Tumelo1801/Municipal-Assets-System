# Use .NET 8 SDK for building
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy csproj and restore dependencies
COPY ["MunicipalAssetsSystem/MunicipalAssetsSystem.csproj", "MunicipalAssetsSystem/"]
RUN dotnet restore "MunicipalAssetsSystem/MunicipalAssetsSystem.csproj"

# Copy everything else and build
COPY . .
WORKDIR "/src/MunicipalAssetsSystem"
RUN dotnet build "MunicipalAssetsSystem.csproj" -c Release -o /app/build

# Publish the app
FROM build AS publish
RUN dotnet publish "MunicipalAssetsSystem.csproj" -c Release -o /app/publish

# Use .NET 8 runtime for running
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS final
WORKDIR /app
COPY --from=publish /app/publish .

# Expose port
EXPOSE 8080

# Set environment variable for port
ENV ASPNETCORE_URLS=http://+:8080

# Run the app
ENTRYPOINT ["dotnet", "MunicipalAssetsSystem.dll"]
